  93def check_permission(required_roles=None, resource_owner_check=False):
  94    """🎍 Decorador que verifica roles y permisos del usuario basado en el JWT.
  95
  96    Args:
  97        required_roles (list, optional): Lista de roles requeridos (ej: ["administrator", "reseller"]).
  98        resource_owner_check (bool): Si True, verifica que el usuario sea dueño del recurso o tenga permisos superiores.
  99
 100    Raises:
 101        Forbidden: Si el usuario no tiene los permisos necesarios.
 102    """
 103    if required_roles is None:
 104        required_roles = []
 105
 106    def decorator(f):
 107        @wraps(f)
 108        def decorated_function(*args, **kwargs):
 109            claims = get_jwt()
 110            user_role = claims.get("rol")
 111            user_id = claims.get("id")
 112
 113            # Verificar si el rol del usuario está en los roles requeridos
 114            if required_roles and user_role not in required_roles:
 115                raise Forbidden("Insufficient permissions for this action.")
 116
 117            # Si se requiere verificación de propietario
 118            if resource_owner_check:
 119                resource_id = kwargs.get("user_id") or kwargs.get("org_id")
 120                if resource_id:
 121                    user = User.query.get(user_id)
 122                    if not user:
 123                        raise NotFound("User not found.")
 124
 125                    # Administradores tienen acceso total
 126                    if user_role == RoleEnum.ADMINISTRATOR.value:
 127                        return f(*args, **kwargs)
 128
 129                    # Verificar si el usuario es el propietario del recurso o tiene permisos superiores
 130                    if user_id != resource_id and not user.is_reseller():
 131                        raise Forbidden("You can only modify your own resources.")
 132
 133            return f(*args, **kwargs)
 134
 135        return decorated_function
 136
 137    return decorator

 140def check_resource_access(resource, claims):
 141    """
 142    Verifica si un usuario tiene acceso a un recurso específico basado en su rol y claims.
 143
 144    Args:
 145        resource: El recurso al que se intenta acceder (debe tener un atributo org_id)
 146        claims: Diccionario con información del usuario (rol, user_id, org_id)
 147
 148    Returns:
 149        bool: True si el usuario tiene acceso, False en caso contrario
 150
 151    Raises:
 152        ValueError: Si los claims no contienen la información necesaria
 153    """
 154    user_role = claims.get("rol")
 155    if not user_role:
 156        return False
 157
 158    # Caso del rol ADMINISTRATOR: acceso total
 159    if user_role == RoleEnum.ADMINISTRATOR.value:
 160        return True
 161
 162    # Caso del rol RESELLER: acceso a recursos de organizaciones en su paquete
 163    if user_role == RoleEnum.RESELLER.value:
 164        reseller_org_id = claims.get("org_id")
 165        if not reseller_org_id:
 166            return False
 167        reseller_package = ResellerPackage.query.filter_by(
 168            reseller_id=reseller_org_id
 169        ).first()
 170        if not reseller_package:
 171            return False
 172        resource_org_id = getattr(resource, "org_id", None)
 173        if resource_org_id is None:
 174            return False
 175        return any(org.id == resource_org_id for org in reseller_package.organizations)
 176
 177    # Caso de roles organizacionales: ORG_ADMIN, ORG_EDITOR, ORG_VIEWER
 178    if user_role in (
 179        RoleEnum.ORG_ADMIN.value,
 180        RoleEnum.ORG_EDITOR.value,
 181        RoleEnum.ORG_VIEWER.value,
 182    ):
 183        user_id = claims.get("user_id")
 184        if not user_id:
 185            return False
 186        user = User.query.get(user_id)
 187        if not user:
 188            return False
 189        resource_org_id = getattr(resource, "org_id", None)
 190        if resource_org_id is None:
 191            return False
 192        return any(org.id == resource_org_id for org in user.organizations)
 193
 194    return False

 201    provided in the JSON body, otherwise ``ip:<ip_address>``. This
 202    covers both:

 205    * dictionary attacks using many usernames from the same IP.

 207    Returns:
 208        str: Limit key for Flask-Limiter.

 211    data = request.get_json(silent=True) or {}
 212    username = (data.get("username") or "").strip().lower()
 213    if username:
 214        return f"ip:{ip}:user:{username}"
 215    return f"ip:{ip}"

 321class LoginView(MethodView):
 322    """Handle user authentication"""
 323
 324    decorators = [
 325        limiter.limit(
 326            f"{CoreConfig.RATE_LIMIT_LOGIN_PER_USER} per {CoreConfig.RATE_LIMIT_LOGIN_WINDOW} seconds",
 327            key_func=_login_limit_key,
 328        ),
 329    ]
 330
 331    def post(self):
 332        """User login endpoint
 333        :param str username: Registered username
 334        :param str password: Account password
 335        :status 200: Successful login
 336        :status 400: Invalid request data
 337        :status 401: Authentication failure
 338        :status 429: Rate limit exceeded
 339        """
 340        try:
 341            data = request.get_json()
 342            if not data:
 343                return jsonify({"msg": "Missing request data"}), 400
 344
 345            username = data.get("username", "").strip()
 346            password = data.get("password", "").strip()
 347
 348            if not username or not password:
 349                return jsonify({"msg": "Credentials required"}), 400
 350
 351            normalized_username = username.lower()
 352
 353            if _is_account_locked(normalized_username):
 354                return jsonify({"msg": "Account temporarily locked"}), 401
 355
 356            user = User.get_by_username(normalized_username)
 357            if self._invalid_credentials(user, password):
 358                # Always perform a dummy hash when user is missing so that
 359                # timing is indistinguishable from a failed password check.
 360                if user is None:
 361                    check_password_hash(
 362                        "pbkdf2:sha256:600000$dummy$dummy",
 363                        password,
 364                    )
 365                _record_failed_login(normalized_username)
 366                return jsonify({"msg": "Invalid credentials"}), 401
 367
 368            _clear_failed_logins(normalized_username)
 369
 370            claims = self._build_claims(user)
 371            tokens = self._generate_tokens(str(user.id), claims)
 372            response = self._build_response(tokens, claims)
 373
 374            return response
 375
 376        except Exception as e:
 377            print(f"Login error: {str(e)}")
 378            return jsonify({"msg": "Invalid request"}), 400
 379
 380    def _invalid_credentials(self, user, password):
 381        """Validate user credentials"""
 382        return not user or not user.check_password(password) or not user.active
 383
 384    def _build_claims(self, user):
 385        """Construct JWT claims payload"""
 386        claims = {
 387            "id": user.id,
 388            "username": user.username,
 389            "email": user.email,
 390            "full_name": user.full_name,
 391            "organizations": [
 392                {"id": org.id, "name": org.name} for org in user.organizations
 393            ],
 394            "rol": user.role.value,
 395        }
 396
 397        if user.role == RoleEnum.RESELLER:
 398            reseller_package = user.reseller_packages.first()
 399            if reseller_package:
 400                claims["reseller_organizations"] = [
 401                    {"id": org.id, "name": org.name}
 402                    for org in reseller_package.organizations
 403                ]
 404        return claims
 405
 406    def _generate_tokens(self, identity, claims):
 407        """Generate JWT tokens"""
 408        return (
 409            create_access_token(identity=identity, additional_claims=claims),
 410            create_refresh_token(identity=identity, additional_claims=claims),
 411        )
 412
 413    def _build_response(self, tokens, claims):
 414        """Build final response with cookies"""
 415        access_token, refresh_token = tokens
 416        response = jsonify(
 417            {
 418                "access_csrf": get_csrf_token(access_token),
 419                "refresh_csrf": get_csrf_token(refresh_token),
 420                "additional_claims": claims,
 421                "msg": "Authentication successful",
 422            }
 423        )
 424        set_access_cookies(response, access_token)
 425        set_refresh_cookies(response, refresh_token)
 426        return response

 456                    "additional_claims": new_claims,

 630    def _update_user(self, user_id, data):
 631        """Actualiza los datos de un usuario existente."""
 632        user = User.query.get_or_404(user_id)
 633        if "username" in data and data["username"] != user.username:
 634            if User.query.filter_by(username=data["username"]).first():
 635                raise BadRequest("Username already exists.")
 636            user.username = data["username"]
 637        if "email" in data and data["email"] != user.email:
 638            if User.query.filter_by(email=data["email"]).first():
 639                raise BadRequest("Email already exists.")
 640            user.email = data["email"]
 641        if "full_name" in data:
 642            user.full_name = data["full_name"]
 643        if "password" in data:
 644            user.set_password(data["password"])
 645        ROLE_MAP = {
 646            "reseller": RoleEnum.RESELLER,
 647            "administrator": RoleEnum.ADMINISTRATOR,
 648            "org_admin": RoleEnum.ORG_ADMIN,
 649            "org_editor": RoleEnum.ORG_EDITOR,
 650            "org_viewer": RoleEnum.ORG_VIEWER,
 651        }
 652        if "role" in data:
 653            claims = get_jwt()
 654            if claims.get("rol") == RoleEnum.ADMINISTRATOR.value:
 655                role = ROLE_MAP.get(data["role"].lower())
 656                if role is None:
 657                    raise BadRequest("Invalid role.")
 658                user.role = role
 659            else:
 660                raise Forbidden("Only administrators can change roles.")
 661        if "organization_id" in data:
 662            organization_id = data["organization_id"]
 663            user = User.query.get(user_id)
 664            if user:
 665                # Desasignar organización anterior si corresponde
 666                if user.organizations:
 667                    for org in user.organizations:
 668                        user.unassign_organization(org.id)
 669                # Asignar nueva organización
 670                user.assign_organization(organization_id)

 673        return jsonify(self._serialize_user(user)), 200
