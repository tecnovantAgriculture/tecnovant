   3"""
   4Image preprocessing and vegetation indices utilities.
   5
   6This module provides:
   7- Color space transforms (sRGB <-> linear).
   8- Simple gray-world white balance.
   9- Robust RGB reading from PNG/JPG/TIFF (8/16-bit, with NoData).
  10- NDVI computation when NIR is available, and visible-only indices
  11  (VARI, NGRDI/GRVI, GLI, ExG) when it is not.
  12- One-shot preprocessing with on-disk caching (PNG preview + NPZ exact floats).
  13- Minimal orchestration that chooses the NDVI path (with NIR) or the
  14  visible-only approximation path (without NIR).
  15
  16Design goals:
  17- Numerical safety: float32 pipelines, NaN propagation, and guarded divisions.
  18- IO robustness: handle NoData, 8/16-bit scaling, and common image formats.
  19- Reproducibility: cache of exact linear RGB in NPZ and a UI-friendly PNG.
  20- Downstream integration: write float32 GeoTIFFs (tiled, compressed) to
  21  interoperate with GIS pipelines without excessive memory.

  24- Inputs and outputs use float32 by default to limit memory footprint.
  25- NDVI ∈ [-1, 1]. Visible-only indices are returned clipped to [-1, 1].

  77def sha256_of_file(fileobj) -> str:
  78    """Compute the SHA-256 hex digest of a file-like object without consuming it."""
  79    fileobj.seek(0)
  80    h = hashlib.sha256()
  81    for chunk in iter(lambda: fileobj.read(8192), b""):
  82        h.update(chunk)
  83    fileobj.seek(0)
  84    return h.hexdigest()

 112def capture_upload_to_temp(file: FileStorage, *, chunk_size: int | None = None) -> UploadCapture:
 113    """Stream ``file`` into a temporary location while hashing and measuring size."""
 114
 115    cfg_chunk = chunk_size or current_app.config.get("MEDIA_UPLOAD_CHUNK_SIZE") or (16 * 1024 * 1024)
 116    chunk_len = max(int(cfg_chunk), 1024 * 1024)
 117    tmp_dir = current_app.config.get("MEDIA_UPLOAD_TMP_DIR")
 118    if not tmp_dir:
 119        tmp_dir = os.path.join(_media_root(), "tmp")
 120    os.makedirs(tmp_dir, exist_ok=True)
 121
 122    stream = file.stream
 123    try:
 124        stream.seek(0)
 125    except Exception:
 126        pass
 127
 128    digest = hashlib.sha256()
 129    size = 0
 130    suffix = Path(file.filename or "").suffix or ".upload"
 131    tmp_path = None
 132
 133    with tempfile.NamedTemporaryFile(prefix="media-upload-", suffix=suffix, dir=tmp_dir, delete=False) as tmp:
 134        while True:
 135            chunk = stream.read(chunk_len)
 136            if not chunk:
 137                break
 138            tmp.write(chunk)
 139            digest.update(chunk)
 140            size += len(chunk)
 141        try:
 142            tmp.flush()
 143            os.fsync(tmp.fileno())
 144        except Exception:
 145            pass
 146        tmp_path = tmp.name
 147
 148    try:
 149        stream.seek(0)
 150    except Exception:
 151        pass
 152
 153    if tmp_path is None:
 154        raise RuntimeError("media: temporary upload path not created")
 155
 156    return UploadCapture(temp_path=tmp_path, size_bytes=size, sha256=digest.hexdigest())

 285def generate_webp_thumbnails(
 286    src_path: str,
 287    asset_uuid: str,
 288    specs: Optional[dict] = None,
 289) -> List[ThumbnailResult]:
 290    """Generate WebP thumbnails for the given image and return metadata.
 291
 292    The default specification produces a single ``gallery`` thumbnail bounded by
 293    512x512 px which maps well to the card width used in the media library grid
 294    (~300 px on desktop) while still yielding a crisp result on HiDPI displays.
 295    The processing pipeline prefers the lightest operations available in Pillow
 296    to keep CPU and memory usage low during high-volume uploads.
 297    """
 298
 299    try:
 300        from PIL import Image, ImageOps  # type: ignore
 301    except Exception:
 302        current_app.logger.warning("Pillow not available; skipping thumbnail generation")
 303        return []
 304
 305    # Default to a single thumbnail tuned for the library grid unless
 306    # configuration overrides are provided.
 307    default_specs = {"gallery": {"max_width": 512, "max_height": 512, "quality": 82, "method": 3}}
 308    specs = specs or current_app.config.get("MEDIA_THUMBNAIL_SPECS", default_specs)
 309
 310    results: List[ThumbnailResult] = []
 311
 312    try:
 313        base_image = Image.open(src_path)
 314        base_image = ImageOps.exif_transpose(base_image)
 315    except Exception:
 316        base_image = None
 317
 318    def build_image_via_rasterio(target_dims: Tuple[int, int]):
 319        try:
 320            import numpy as np  # type: ignore
 321            import rasterio  # type: ignore
 322            from rasterio.enums import Resampling  # type: ignore
 323        except Exception:
 324            return None
 325
 326        try:
 327            with rasterio.open(src_path) as src:
 328                band_count = max(1, min(3, src.count))
 329                indexes = list(range(1, band_count + 1))
 330                scale = max(src.width / target_dims[0], src.height / target_dims[1], 1)
 331                out_w = max(1, int(src.width / scale))
 332                out_h = max(1, int(src.height / scale))
 333                data = src.read(
 334                    indexes=indexes,
 335                    out_shape=(band_count, out_h, out_w),
 336                    resampling=Resampling.bilinear,
 337                )
 338        except Exception:
 339            return None
 340
 341        try:
 342            data = data.astype("float32")
 343            data = np.nan_to_num(data)
 344            min_val = data.min()
 345            data -= min_val
 346            max_val = data.max()
 347            if max_val > 0:
 348                data /= max_val
 349            data = (data * 255).clip(0, 255).astype("uint8")
 350            if band_count == 1:
 351                data = np.repeat(data, 3, axis=0)
 352            data = np.moveaxis(data, 0, -1)
 353            return Image.fromarray(data)
 354        except Exception:
 355            return None
 356
 357    for kind, cfg in specs.items():
 358        target = (cfg.get("max_width", 512), cfg.get("max_height", 512))
 359        quality = cfg.get("quality", 82)
 360        method = cfg.get("method", 3)
 361        storage_key, abs_path = allocate_variant_path(asset_uuid, kind)
 362
 363        im = base_image.copy() if base_image else build_image_via_rasterio(target)
 364        if im is None:
 365            current_app.logger.exception("Failed to build %s thumbnail for asset %s", kind, asset_uuid)
 366            continue
 367
 368        try:
 369            # Pillow keeps data lazy-loaded; ``draft`` hints reduce memory for JPEGs.
 370            try:
 371                im.draft(im.mode, target)
 372            except Exception:
 373                pass
 374
 375            if im.mode not in ("RGB", "RGBA"):
 376                im = im.convert("RGB")
 377
 378            resampling_attr = getattr(Image, "Resampling", None)
 379            if resampling_attr is not None:
 380                resample_filter = resampling_attr.BILINEAR
 381            else:
 382                resample_filter = Image.BILINEAR
 383            im.thumbnail(target, resample=resample_filter)
 384
 385            save_kwargs = {
 386                "format": "WEBP",
 387                "quality": quality,
 388                "method": method,
 389                "optimize": True,
 390            }
 391
 392            im.save(abs_path, **save_kwargs)
 393            width, height = im.size
 394        except Exception:
 395            current_app.logger.exception("Failed to build %s thumbnail for asset %s", kind, asset_uuid)
 396            continue
 397
 398        results.append(ThumbnailResult(kind=kind, storage_key=storage_key, width=width, height=height))
 399
 400    if base_image is not None:
 401        try:
 402            base_image.close()
 403        except Exception:
 404            pass
 405
 406    return results

 410def robust_minmax(arr: np.ndarray, mask: np.ndarray | None = None, p_lo=2.0, p_hi=98.0):
 411    """Compute robust min/max percentiles optionally ignoring masked pixels."""
 412    if mask is not None:
 413        data = arr[~mask]
 414    else:
 415        data = arr.ravel()
 416    if data.size == 0:
 417        return 0.0, 1.0
 418    lo = np.percentile(data, p_lo)
 419    hi = np.percentile(data, p_hi)
 420    if hi <= lo:
 421        hi = lo + 1.0
 422    return float(lo), float(hi)

 425def normalize_block(band: np.ndarray, lo: float, hi: float) -> np.ndarray:
 426    """Normalize a single band with precomputed min/max, clipping to [0, 1]."""
 427    x = (band.astype("float32") - lo) / (hi - lo + 1e-8)
 428    return np.clip(x, 0.0, 1.0)

 431def read_rgb_block(src, win: Window, band_map=(1, 2, 3)):
 432    """Read a small RGB window from ``src`` using ``band_map`` ordering."""
 433    r = src.read(band_map[0], window=win)
 434    g = src.read(band_map[1], window=win)
 435    b = src.read(band_map[2], window=win)
 436    return r, g, b

 439def compute_block_minmax(src, band_map=(1, 2, 3), block=1024):
 440    """Scan an image in blocks to estimate robust min/max for each RGB band."""
 441    H, W = src.height, src.width
 442    r_vals, g_vals, b_vals = [], [], []
 443    for y in range(0, H, block):
 444        for x in range(0, W, block):
 445            win = Window(x, y, min(block, W - x), min(block, H - y))
 446            r, g, b = read_rgb_block(src, win, band_map)
 447            mask = None
 448            if src.nodata is not None:
 449                mask = (r == src.nodata) | (g == src.nodata) | (b == src.nodata)
 450            rl, rh = robust_minmax(r, mask)
 451            gl, gh = robust_minmax(g, mask)
 452            bl, bh = robust_minmax(b, mask)
 453            r_vals.append((rl, rh)); g_vals.append((gl, gh)); b_vals.append((bl, bh))
 454    r_lo = float(np.median([v[0] for v in r_vals])); r_hi = float(np.median([v[1] for v in r_vals]))
 455    g_lo = float(np.median([v[0] for v in g_vals])); g_hi = float(np.median([v[1] for v in g_vals]))
 456    b_lo = float(np.median([v[0] for v in b_vals])); b_hi = float(np.median([v[1] for v in b_vals]))
 457    return (r_lo, r_hi), (g_lo, g_hi), (b_lo, b_hi)

 481def srgb_to_linear(x: NDArray[np.floating]) -> NDArray[np.float32]:
 482    """
 483    Convert sRGB to linear RGB (gamma removal).
 484
 485    Parameters
 486    ----------
 487    x
 488        Array-like of sRGB values in [0, 1]. Any shape, last dimension length
 489        may be 3 for RGB but is not required. Values outside [0, 1] are not
 490        clamped here to avoid hiding data issues upstream.
 491
 492    Returns
 493    -------
 494    numpy.ndarray
 495        Linear RGB as float32, same shape as `x`.
 496
 497    Notes
 498    -----
 499    Uses the standard IEC 61966-2-1 transfer function. Thresholds and constants:
 500    - a = 0.055
 501    - thr = 0.04045
 502
 503    This function operates vectorized and preserves NaN locations.
 504    """
 505    a, thr = 0.055, 0.04045
 506    y = np.empty_like(x, dtype=np.float32)
 507    low = x <= thr
 508    y[low] = x[low] / 12.92
 509    y[~low] = ((x[~low] + a) / (1.0 + a)) ** 2.4
 510    return y.astype(np.float32)

 513def linear_to_srgb(x: NDArray[np.floating]) -> NDArray[np.float32]:
 514    """
 515    Convert linear RGB to sRGB (gamma application).
 516
 517    Parameters
 518    ----------
 519    x
 520        Array-like of linear RGB values, typically in [0, 1]. Any shape.
 521
 522    Returns
 523    -------
 524    numpy.ndarray
 525        sRGB as float32 in [0, 1], same shape as `x`. Values are clipped to
 526        [0, 1] at the end.
 527
 528    Notes
 529    -----
 530    Uses the inverse IEC 61966-2-1 transfer function. Thresholds and constants:
 531    - a = 0.055
 532    - thr = 0.0031308
 533    """
 534    a, thr = 0.055, 0.0031308
 535    y = np.empty_like(x, dtype=np.float32)
 536    low = x <= thr
 537    y[low] = 12.92 * x[low]
 538    y[~low] = (1 + a) * (x[~low]) ** (1 / 2.4) - a
 539    return np.clip(y, 0.0, 1.0).astype(np.float32)

 542def gray_world(img_lin: NDArray[np.floating]) -> NDArray[np.float32]:
 543    """
 544    Apply Gray-World white balance to linear RGB.
 545
 546    Parameters
 547    ----------
 548    img_lin
 549        Linear RGB image as float array with last dimension size 3.
 550        Expected range [0, 1], NaN allowed (they are ignored in scaling).
 551
 552    Returns
 553    -------
 554    numpy.ndarray
 555        White-balanced linear RGB as float32, clipped to [0, 1].
 556
 557    Notes
 558    -----
 559    The Gray-World assumption scales each channel by the ratio of the mean of
 560    all channels to the channel mean: scale_c = mean(mean(R,G,B)) / mean(c).
 561    A small epsilon (1e-8) avoids division by zero.
 562    """
 563    if img_lin.shape[-1] != 3:
 564        raise ValueError("img_lin must have last dimension of size 3 (RGB).")
 565    means = img_lin.reshape(-1, 3).mean(axis=0)
 566    scale = means.mean() / (means + 1e-8)
 567    return np.clip(img_lin * scale, 0.0, 1.0).astype(np.float32)

 594def read_rgb(path: Path) -> NDArray[np.float32]:
 595    """
 596    Read an image and return sRGB float32 in [0, 1].
 597
 598    Parameters
 599    ----------
 600    path
 601        Path to PNG/JPG/TIFF. TIFF may be 8/16-bit and can carry NoData.
 602
 603    Returns
 604    -------
 605    numpy.ndarray
 606        Array of shape (H, W, 3) float32 in [0, 1]. Pixels matching NoData in
 607        any RGB band are returned as NaN triplets.
 608
 609    Behavior
 610    --------
 611    1) Try rasterio:
 612       - Read bands 1..3 as float32.
 613       - Scale by dtype max if > 1 (e.g., 65535 for uint16).
 614       - Apply NoData as NaN.
 615    2) Fallback to PIL:
 616       - Convert to 'RGB' and scale by 255.
 617
 618    Raises
 619    ------
 620    FileNotFoundError
 621        If the file does not exist.
 622    ValueError
 623        If a raster is opened but has fewer than 3 bands.
 624
 625    Notes
 626    -----
 627    The function prefers rasterio to preserve georeferencing metadata when
 628    available, though it returns only pixel data.
 629    """
 630    if not Path(path).exists():
 631        raise FileNotFoundError(f"File not found: {path}")
 632
 633    try:
 634        with rasterio.open(path) as src:
 635            if src.count < 3:
 636                raise ValueError("Raster has fewer than 3 bands.")
 637            r = src.read(1).astype(np.float32)
 638            g = src.read(2).astype(np.float32)
 639            b = src.read(3).astype(np.float32)
 640            maxv = _dtype_max(src.dtypes[0])
 641            if maxv > 1.0:
 642                r, g, b = r / maxv, g / maxv, b / maxv
 643            rgb = np.stack([r, g, b], axis=-1).astype(np.float32)
 644            if src.nodata is not None:
 645                nod = src.nodata
 646                mask = (r == nod) | (g == nod) | (b == nod)
 647                rgb[mask] = np.nan
 648            return rgb.astype(np.float32)
 649    except Exception:
 650        # Fallback for non-GDAL formats or simple images.
 651        im = Image.open(path).convert("RGB")
 652        return (np.asarray(im, dtype=np.float32) / 255.0).astype(np.float32)

 655def read_red_nir(
 656    src: DatasetReader,
 657    red_band: int = 3,
 658    nir_band: int = 4,
 659) -> Tuple[NDArray[np.float32], NDArray[np.float32]]:
 660    """
 661    Read and normalize RED and NIR bands from an open rasterio dataset.
 662
 663    Parameters
 664    ----------
 665    src
 666        Open rasterio dataset.
 667    red_band
 668        1-based index of the RED band (default 3).
 669    nir_band
 670        1-based index of the NIR band (default 4).
 671
 672    Returns
 673    -------
 674    (red, nir)
 675        Tuple of float32 arrays in [0, 1] when integer input types are used.
 676        NoData values are returned as NaN.
 677
 678    Raises
 679    ------
 680    ValueError
 681        If band indices are invalid or dataset has insufficient bands.
 682    """
 683    if src.count < max(red_band, nir_band):
 684        raise ValueError("Dataset does not contain the requested bands.")
 685    red = src.read(red_band).astype(np.float32)
 686    nir = src.read(nir_band).astype(np.float32)
 687    maxv = _dtype_max(src.dtypes[0])
 688    if maxv > 1.0:
 689        red, nir = red / maxv, nir / maxv
 690    if src.nodata is not None:
 691        m = (red == src.nodata) | (nir == src.nodata)
 692        red[m] = np.nan
 693        nir[m] = np.nan
 694    return red.astype(np.float32), nir.astype(np.float32)

 697def save_quick_preview(
 698    src_path: Path,
 699    out_path: Path,
 700    max_size: int = 1024,
 701) -> Tuple[int, int]:
 702    """Generate a fast RGB preview PNG from the source image.
 703
 704    This function replicates the legacy Agrovista preview pipeline:
 705    1. Read RGB (handling 8/16 bit, NoData).
 706    2. Replace NaN with 0 (black).
 707    3. Clip to [0, 1].
 708    4. Scale to uint8 [0, 255].
 709    5. Resize (thumbnail) to max_size.
 710    6. Save as PNG.
 711    """
 712    # Use media's read_rgb which is robust for 8/16 bit and NoData
 713    rgb = read_rgb(src_path)
 714
 715    # Handle NaN -> 0 (black) and clip
 716    arr = np.nan_to_num(rgb, nan=0.0)
 717    arr = np.clip(arr, 0.0, 1.0)
 718
 719    # Convert to uint8 image
 720    im = Image.fromarray((arr * 255).astype(np.uint8))
 721
 722    # Resize if needed
 723    w, h = im.size
 724    if max(w, h) > max_size:
 725        im.thumbnail((max_size, max_size))
 726
 727    im.save(out_path)
 728    return im.size

 808def preprocess_rgb_once(
 809    input_path: Path,
 810    cfg: PreprocessConfig,
 811) -> Tuple[
 812    NDArray[np.float32],
 813    Optional[Path],
 814    Optional[Path],
 815    Optional[Path],
 816    Optional[Path],
 817]:
 818    """
 819    Preprocess RGB exactly once and cache results to disk.
 820
 821    Steps
 822    -----
 823    1) Read sRGB [0, 1] and convert to linear RGB.
 824    2) Optional gray-world white balance.
 825    3) Optional shadow mask: pixels with low linear RGB sum are set to NaN.
 826    4) Cache artifacts:
 827       - NPZ with exact linear float32 RGB (for scientific use).
 828       - PNG (sRGB 8-bit) preview for UI.
 829
 830    Parameters
 831    ----------
 832    input_path
 833        Path to the input image (PNG/JPG/TIFF).
 834    cfg
 835        PreprocessConfig controlling the operations and cache directory.
 836
 837    Returns
 838    -------
 839    (rgb_lin, png_path, npz_path, vi_gray_path, vi_heat_path, heatmap_path)
 840        - rgb_lin: linear RGB float32 en [0, 1] con NaNs donde aplique.
 841        - png_path: vista previa RGB normalizada en PNG, si se generó.
 842        - npz_path: cache NPZ con los flotantes exactos, si se generó.
 843        - vi_gray_path: PNG en escala de grises con VI (G/R), si se generó.
 844        - vi_heat_path: PNG coloreado (tonos naranja/rojo) derivado del VI.
 845        - heatmap_path: PNG RGBA con heatmap pseudo-NDVI y máscara de vegetación.
 846
 847    Notes
 848    -----
 849    - If the NPZ cache already exists, it is used directly (fast path). The
 850      preview PNG is not recomputed if already present.
 851    - NaN pixels are mapped to black (0) in the preview PNG to avoid visible
 852      checkerboards, but the NPZ preserves NaNs exactly.
 853    """
 854    in_key = input_path.stem
 855    cache_dir = cfg.cache_dir or input_path.parent
 856    cache_dir.mkdir(parents=True, exist_ok=True)
 857    npz_path = cache_dir / f"{in_key}__rgb_preproc_linear.npz"
 858    png_path = cache_dir / f"{in_key}__rgb_preproc_preview.png"
 859
 860    def _downsample_for_preview(rgb_data: NDArray[np.float32], max_dim: int) -> NDArray[np.float32]:
 861        h, w = rgb_data.shape[:2]
 862        if max_dim <= 0 or max(h, w) <= max_dim:
 863            return rgb_data
 864        step = int(np.ceil(max(h, w) / max_dim))
 865        step = max(1, step)
 866        ds = rgb_data[::step, ::step]
 867        if ds.size == 0:
 868            return rgb_data
 869        # Ensure we do not exceed max_dim due to rounding.
 870        h2, w2 = ds.shape[:2]
 871        if max(h2, w2) > max_dim:
 872            step2 = int(np.ceil(max(h2, w2) / max_dim))
 873            step2 = max(1, step2)
 874            ds = ds[::step2, ::step2]
 875            if ds.size == 0:
 876                return rgb_data
 877        return ds
 878
 879    def _apply_heatmap_colormap(norm: NDArray[np.float32], mask: Optional[NDArray[np.bool_]]) -> NDArray[np.uint8]:
 880        stops = np.array([0.0, 0.5, 1.0], dtype=np.float32)
 881        colors = np.array(
 882            [
 883                [0.647, 0.0, 0.149],   # deep red
 884                [1.0, 0.973, 0.533],   # yellow
 885                [0.0, 0.4, 0.0],       # dark green
 886            ],
 887            dtype=np.float32,
 888        )
 889        flat = norm.ravel()
 890        r = np.interp(flat, stops, colors[:, 0]).reshape(norm.shape)
 891        g = np.interp(flat, stops, colors[:, 1]).reshape(norm.shape)
 892        b = np.interp(flat, stops, colors[:, 2]).reshape(norm.shape)
 893        rgba = np.stack(
 894            [
 895                (r * 255.0).astype(np.uint8),
 896                (g * 255.0).astype(np.uint8),
 897                (b * 255.0).astype(np.uint8),
 898                np.full(norm.shape, 255, dtype=np.uint8),
 899            ],
 900            axis=-1,
 901        )
 902        if mask is not None:
 903            alpha = np.where(mask, 255, 0).astype(np.uint8)
 904            rgba[..., 3] = alpha
 905        return rgba
 906
 907    def ensure_rgb_preview() -> Optional[Path]:
 908        if png_path.exists():
 909            return png_path
 910        save_quick_preview(input_path, png_path, max_size=cfg.preview_max_dim)
 911        return png_path
 912
 913    def ensure_vi_outputs(rgb_data: NDArray[np.float32]) -> Tuple[Optional[Path], Optional[Path]]:
 914        vi_gray_path = cache_dir / f"{in_key}__vi_gr_ratio.png"
 915        vi_heat_path = cache_dir / f"{in_key}__vi_gr_heat.png"
 916        if vi_gray_path.exists() and vi_heat_path.exists():
 917            return vi_gray_path, vi_heat_path
 918
 919        vi_src = _downsample_for_preview(rgb_data, cfg.preview_max_dim)
 920
 921        g = vi_src[..., 1]
 922        r = vi_src[..., 0]
 923        vi_ratio = np.divide(g, r + 1e-6)
 924        vi_ratio[np.isnan(vi_src).any(axis=-1)] = np.nan
 925
 926        if not np.isfinite(vi_ratio).any():
 927            return None, None
 928
 929        lo = float(np.nanpercentile(vi_ratio, 5))
 930        hi = float(np.nanpercentile(vi_ratio, 95))
 931        if not np.isfinite(lo):
 932            lo = float(np.nanmin(vi_ratio))
 933        if not np.isfinite(hi):
 934            hi = float(np.nanmax(vi_ratio))
 935        if hi <= lo:
 936            hi = lo + 1e-3
 937        vi_norm = np.clip((vi_ratio - lo) / (hi - lo), 0.0, 1.0)
 938        vi_norm = np.nan_to_num(vi_norm, nan=0.0)
 939
 940        vi_gray = (vi_norm * 255.0).astype(np.uint8)
 941        Image.fromarray(vi_gray, mode="L").save(vi_gray_path)
 942
 943        heat = np.zeros((*vi_norm.shape, 3), dtype=np.float32)
 944        heat[..., 0] = vi_norm
 945        heat[..., 1] = np.clip(vi_norm * 0.6, 0.0, 1.0)
 946        heat[..., 2] = np.clip(vi_norm * 0.2, 0.0, 1.0)
 947        heat_img = (heat * 255.0).astype(np.uint8)
 948        Image.fromarray(heat_img, mode="RGB").save(vi_heat_path)
 949
 950        return vi_gray_path, vi_heat_path
 951
 952    def ensure_heatmap(rgb_data: NDArray[np.float32]) -> Optional[Path]:
 953        heatmap_path = cache_dir / f"{in_key}__vi_heatmap.png"
 954        if heatmap_path.exists():
 955            return heatmap_path
 956
 957        heat_src = _downsample_for_preview(rgb_data, cfg.preview_max_dim)
 958        R = heat_src[..., 0]
 959        G = heat_src[..., 1]
 960        B = heat_src[..., 2]
 961        EPS = 1e-6
 962
 963        ngrdi = (G - R) / (G + R + EPS)
 964        vari = (G - R) / (G + R - B + EPS)
 965        gli = (2.0 * G - R - B) / (2.0 * G + R + B + EPS)
 966        exg = 2.0 * G - R - B
 967        pseudo = (
 968            0.4 * ngrdi
 969            + 0.25 * vari
 970            + 0.25 * gli
 971            + 0.10 * exg
 972        )
 973        pseudo = np.clip(pseudo, -1.0, 1.0).astype(np.float32)
 974
 975        vi = G / (R + EPS)
 976        mask = (exg > 0.05) & (ngrdi > 0.0) & (vi > 1.2)
 977
 978        norm = (pseudo + 1.0) * 0.5
 979        norm = np.clip(norm, 0.0, 1.0).astype(np.float32)
 980
 981        rgba = _apply_heatmap_colormap(norm, mask)
 982        try:
 983            Image.fromarray(rgba, mode="RGBA").save(heatmap_path)
 984        except Exception:
 985            return None
 986        return heatmap_path
 987
 988    cached_rgb: Optional[NDArray[np.float32]] = None
 989    if npz_path.exists():
 990        try:
 991            data = np.load(npz_path)
 992            cached_rgb = data["rgb"].astype(np.float32)
 993        except Exception:
 994            try:
 995                npz_path.unlink()
 996            except OSError:
 997                pass
 998            try:
 999                png_path.unlink()
1000            except OSError:
1001                pass
1002            vi_cache = cache_dir / f"{in_key}__vi_gr_ratio.png"
1003            try:
1004                vi_cache.unlink()
1005            except OSError:
1006                pass
1007            vi_heat_cache = cache_dir / f"{in_key}__vi_gr_heat.png"
1008            try:
1009                vi_heat_cache.unlink()
1010            except OSError:
1011                pass
1012            heatmap_cache = cache_dir / f"{in_key}__vi_heatmap.png"
1013            try:
1014                heatmap_cache.unlink()
1015            except OSError:
1016                pass
1017            current_app.logger.warning(
1018                "media: invalid NPZ cache detected for %s; regenerating artifacts",
1019                input_path,
1020            )
1021        else:
1022            preview_path = ensure_rgb_preview()
1023            vi_gray_path, vi_heat_path = ensure_vi_outputs(cached_rgb)
1024            heatmap_path = ensure_heatmap(cached_rgb)
1025            return cached_rgb, preview_path, npz_path, vi_gray_path, vi_heat_path, heatmap_path
1026
1027    rgb_srgb = read_rgb(input_path).astype(np.float32)
1028    nan_mask = np.isnan(rgb_srgb).any(axis=-1)
1029    rgb_srgb[nan_mask] = 0.0  # keep mask but avoid NaN propagation in transfer
1030    rgb_lin = srgb_to_linear(rgb_srgb)
1031
1032    if cfg.apply_gray_world:
1033        rgb_lin = gray_world(rgb_lin)
1034
1035    if cfg.apply_shadow_mask:
1036        dark = (rgb_lin[..., 0] + rgb_lin[..., 1] + rgb_lin[..., 2]) < cfg.shadow_thr
1037        rgb_lin[dark] = np.nan
1038
1039    np.savez_compressed(npz_path, rgb=rgb_lin.astype(np.float32))
1040
1041    preview_path = ensure_rgb_preview()
1042    vi_gray_path, vi_heat_path = ensure_vi_outputs(rgb_lin)
1043    heatmap_path = ensure_heatmap(rgb_lin)
1044
1045    return rgb_lin.astype(np.float32), preview_path, npz_path, vi_gray_path, vi_heat_path, heatmap_path

1051def visible_indices(
1052    rgb_lin: NDArray[np.floating],
1053) -> Dict[str, NDArray[np.float32]]:
1054    """
1055    Compute visible-only vegetation indices from linear RGB.
1056
1057    Parameters
1058    ----------
1059    rgb_lin
1060        Linear RGB image (H, W, 3) in [0, 1]. NaNs are allowed and propagate.
1061
1062    Returns
1063    -------
1064    dict
1065        Dictionary with float32 arrays in [-1, 1]:
1066        - "VARI"  : (G - R) / (G + R - B + eps)
1067        - "NGRDI" : (G - R) / (G + R + eps)  (also called GRVI)
1068        - "GLI"   : (2G - R - B) / (2G + R + B + eps)
1069        - "ExG"   : 2*gN - rN - bN, with rN,gN,bN normalized by (R+G+B)
1070
1071    Notes
1072    -----
1073    - Uses eps=1e-6 to guard divisions.
1074    - Returns values clipped to [-1, 1]; NaNs are preserved.
1075    """
1076    if rgb_lin.shape[-1] != 3:
1077        raise ValueError("rgb_lin must have last dimension of size 3 (RGB).")
1078    r, g, b = rgb_lin[..., 0], rgb_lin[..., 1], rgb_lin[..., 2]
1079    eps = 1e-6
1080    with np.errstate(divide="ignore", invalid="ignore"):
1081        vari = (g - r) / (g + r - b + eps)
1082        ngrdi = (g - r) / (g + r + eps)
1083        gli = (2.0 * g - r - b) / (2.0 * g + r + b + eps)
1084        total = r + g + b
1085        r_n, g_n, b_n = r / (total + eps), g / (total + eps), b / (total + eps)
1086        exg = 2.0 * g_n - r_n - b_n
1087    return {
1088        "VARI": np.clip(vari, -1.0, 1.0).astype(np.float32),
1089        "NGRDI": np.clip(ngrdi, -1.0, 1.0).astype(np.float32),
1090        "GLI": np.clip(gli, -1.0, 1.0).astype(np.float32),
1091        "ExG": np.clip(exg, -1.0, 1.0).astype(np.float32),
1092    }

1095def combine_indices(
1096    indices: Dict[str, NDArray[np.floating]],
1097    method: str = "combined",
1098    weights: Tuple[float, float, float, float] = (0.4, 0.3, 0.2, 0.1),
1099) -> NDArray[np.float32]:
1100    """
1101    Combine visible indices or select a single index.
1102
1103    Parameters
1104    ----------
1105    indices
1106        Dict as returned by `visible_indices`.
1107    method
1108        One of {"combined", "ngrdi", "vari", "gli", "exg"} (case-sensitive).
1109    weights
1110        Weights for ("NGRDI", "VARI", "GLI", "ExG") when `method="combined"`.
1111
1112    Returns
1113    -------
1114    numpy.ndarray
1115        Float32 array in [-1, 1].
1116
1117    Raises
1118    ------
1119    KeyError
1120        If required indices are missing for the chosen method.
1121    ValueError
1122        If `method` is not recognized.
1123
1124    Notes
1125    -----
1126    The default combination emphasizes NGRDI and VARI, which often show good
1127    contrast for vegetation in visible-only scenarios.
1128    """
1129    if method == "ngrdi":
1130        return indices["NGRDI"].astype(np.float32)
1131    if method == "vari":
1132        return indices["VARI"].astype(np.float32)
1133    if method == "gli":
1134        return indices["GLI"].astype(np.float32)
1135    if method == "exg":
1136        return indices["ExG"].astype(np.float32)
1137    if method != "combined":
1138        raise ValueError(f"Unknown method: {method}")
1139
1140    out = (
1141        weights[0] * indices["NGRDI"]
1142        + weights[1] * indices["VARI"]
1143        + weights[2] * indices["GLI"]
1144        + weights[3] * indices["ExG"]
1145    )
1146    return np.clip(out, -1.0, 1.0).astype(np.float32)

1149def true_ndvi(
1150    src: DatasetReader,
1151    red_band: int = 3,
1152    nir_band: int = 4,
1153) -> NDArray[np.float32]:
1154    """
1155    Compute true NDVI from an open rasterio dataset.
1156
1157    Parameters
1158    ----------
1159    src
1160        Open dataset with RED and NIR bands.
1161    red_band
1162        1-based index of the RED band.
1163    nir_band
1164        1-based index of the NIR band.
1165
1166    Returns
1167    -------
1168    numpy.ndarray
1169        NDVI float32 in [-1, 1]. NaNs propagate.
1170
1171    Formula
1172    -------
1173    NDVI = (NIR - RED) / (NIR + RED)
1174
1175    Notes
1176    -----
1177    Division is guarded: when (NIR + RED) == 0, the output is NaN.
1178    """
1179    red, nir = read_red_nir(src, red_band=red_band, nir_band=nir_band)
1180    denom = nir + red
1181    ndvi = np.divide(
1182        nir - red,
1183        denom,
1184        out=np.full_like(red, np.nan, dtype=np.float32),
1185        where=denom != 0,
1186    )
1187    return ndvi.astype(np.float32)
