   4Image preprocessing and vegetation indices utilities.

   6This module provides:
   7- Color space transforms (sRGB <-> linear).

  78def sha256_of_file(fileobj) -> str:
  79    """Compute the SHA-256 hex digest of a file-like object without consuming it."""
  80    fileobj.seek(0)
  81    h = hashlib.sha256()
  82    for chunk in iter(lambda: fileobj.read(8192), b""):
  83        h.update(chunk)
  84    fileobj.seek(0)
  85    return h.hexdigest()

 115def capture_upload_to_temp(
 116    file: FileStorage, *, chunk_size: int | None = None
 117) -> UploadCapture:
 118    """Stream ``file`` into a temporary location while hashing and measuring size."""
 119
 120    cfg_chunk = (
 121        chunk_size
 122        or current_app.config.get("MEDIA_UPLOAD_CHUNK_SIZE")
 123        or (16 * 1024 * 1024)
 124    )
 125    chunk_len = max(int(cfg_chunk), 1024 * 1024)
 126    tmp_dir = current_app.config.get("MEDIA_UPLOAD_TMP_DIR")
 127    if not tmp_dir:
 128        tmp_dir = os.path.join(_media_root(), "tmp")
 129    os.makedirs(tmp_dir, exist_ok=True)
 130
 131    stream = file.stream
 132    try:
 133        stream.seek(0)
 134    except Exception:
 135        pass
 136
 137    digest = hashlib.sha256()
 138    size = 0
 139    suffix = Path(file.filename or "").suffix or ".upload"
 140    tmp_path = None
 141
 142    with tempfile.NamedTemporaryFile(
 143        prefix="media-upload-", suffix=suffix, dir=tmp_dir, delete=False
 144    ) as tmp:
 145        while True:
 146            chunk = stream.read(chunk_len)
 147            if not chunk:
 148                break
 149            tmp.write(chunk)
 150            digest.update(chunk)
 151            size += len(chunk)
 152        try:
 153            tmp.flush()
 154            if os.getenv("MEDIA_UPLOAD_FSYNC", "false").lower() == "true":
 155                os.fsync(tmp.fileno())
 156        except Exception:
 157            pass
 158        tmp_path = tmp.name
 159
 160    try:
 161        stream.seek(0)
 162    except Exception:
 163        pass
 164
 165    if tmp_path is None:
 166        raise RuntimeError("media: temporary upload path not created")
 167
 168    return UploadCapture(temp_path=tmp_path, size_bytes=size, sha256=digest.hexdigest())

 307def generate_webp_thumbnails(
 308    src_path: str,
 309    asset_uuid: str,
 310    specs: Optional[dict] = None,
 311) -> List[ThumbnailResult]:
 312    """Generate WebP thumbnails for the given image and return metadata.
 313
 314    The default specification produces a single ``gallery`` thumbnail bounded by
 315    512x512 px which maps well to the card width used in the media library grid
 316    (~300 px on desktop) while still yielding a crisp result on HiDPI displays.
 317    The processing pipeline prefers the lightest operations available in Pillow
 318    to keep CPU and memory usage low during high-volume uploads.
 319    """
 320
 321    try:
 322        from PIL import Image, ImageOps  # type: ignore
 323    except Exception:
 324        current_app.logger.warning(
 325            "Pillow not available; skipping thumbnail generation"
 326        )
 327        return []
 328
 329    # Default to a single thumbnail tuned for the library grid unless
 330    # configuration overrides are provided.
 331    default_specs = {
 332        "gallery": {"max_width": 512, "max_height": 512, "quality": 82, "method": 3}
 333    }
 334    specs = specs or current_app.config.get("MEDIA_THUMBNAIL_SPECS", default_specs)
 335
 336    results: List[ThumbnailResult] = []
 337
 338    def build_image_via_rasterio(target_dims: Tuple[int, int]):
 339        try:
 340            import numpy as np  # type: ignore
 341            import rasterio  # type: ignore
 342            from rasterio.enums import Resampling  # type: ignore
 343        except Exception:
 344            return None
 345
 346        try:
 347            with rasterio.open(src_path) as src:
 348                band_count = max(1, min(3, src.count))
 349                indexes = list(range(1, band_count + 1))
 350                scale = max(src.width / target_dims[0], src.height / target_dims[1], 1)
 351                out_w = max(1, int(src.width / scale))
 352                out_h = max(1, int(src.height / scale))
 353                data = src.read(
 354                    indexes=indexes,
 355                    out_shape=(band_count, out_h, out_w),
 356                    resampling=Resampling.bilinear,
 357                )
 358        except Exception:
 359            return None
 360
 361        try:
 362            data = data.astype("float32")
 363            data = np.nan_to_num(data)
 364            min_val = data.min()
 365            data -= min_val
 366            max_val = data.max()
 367            if max_val > 0:
 368                data /= max_val
 369            data = (data * 255).clip(0, 255).astype("uint8")
 370            if band_count == 1:
 371                data = np.repeat(data, 3, axis=0)
 372            data = np.moveaxis(data, 0, -1)
 373            return Image.fromarray(data)
 374        except Exception:
 375            return None
 376
 377    def build_image_via_pillow():
 378        try:
 379            im = Image.open(src_path)
 380            im = ImageOps.exif_transpose(im)
 381            return im
 382        except Exception:
 383            return None
 384
 385    for kind, cfg in specs.items():
 386        target = (cfg.get("max_width", 512), cfg.get("max_height", 512))
 387        quality = cfg.get("quality", 82)
 388        method = cfg.get("method", 3)
 389        storage_key, abs_path = allocate_variant_path(asset_uuid, kind)
 390
 391        # Rasterio-first: handles GeoTIFFs at output resolution without loading
 392        # the full raster into RAM. Falls back to Pillow for formats rasterio
 393        # cannot open (plain PNG/JPG without geo metadata).
 394        im = build_image_via_rasterio(target) or build_image_via_pillow()
 395        if im is None:
 396            current_app.logger.exception(
 397                "Failed to build %s thumbnail for asset %s", kind, asset_uuid
 398            )
 399            continue
 400
 401        try:
 402            # Pillow keeps data lazy-loaded; ``draft`` hints reduce memory for JPEGs.
 403            try:
 404                im.draft(im.mode, target)
 405            except Exception:
 406                pass
 407
 408            if im.mode not in ("RGB", "RGBA"):
 409                im = im.convert("RGB")
 410
 411            resampling_attr = getattr(Image, "Resampling", None)
 412            if resampling_attr is not None:
 413                resample_filter = resampling_attr.BILINEAR
 414            else:
 415                resample_filter = Image.BILINEAR
 416            im.thumbnail(target, resample=resample_filter)
 417
 418            save_kwargs = {
 419                "format": "WEBP",
 420                "quality": quality,
 421                "method": method,
 422                "optimize": True,
 423            }
 424
 425            im.save(abs_path, **save_kwargs)
 426            width, height = im.size
 427        except Exception:
 428            current_app.logger.exception(
 429                "Failed to build %s thumbnail for asset %s", kind, asset_uuid
 430            )
 431            continue
 432
 433        results.append(
 434            ThumbnailResult(
 435                kind=kind, storage_key=storage_key, width=width, height=height
 436            )
 437        )
 438
 439    return results

 442def robust_minmax(arr: np.ndarray, mask: np.ndarray | None = None, p_lo=2.0, p_hi=98.0):
 443    """Compute robust min/max percentiles optionally ignoring masked pixels."""
 444    if mask is not None:
 445        data = arr[~mask]
 446    else:
 447        data = arr.ravel()
 448    if data.size == 0:
 449        return 0.0, 1.0
 450    lo = np.percentile(data, p_lo)
 451    hi = np.percentile(data, p_hi)
 452    if hi <= lo:
 453        hi = lo + 1.0
 454    return float(lo), float(hi)

 457def normalize_block(band: np.ndarray, lo: float, hi: float) -> np.ndarray:
 458    """Normalize a single band with precomputed min/max, clipping to [0, 1]."""

 463def read_rgb_block(src, win: Window, band_map=(1, 2, 3)):
 464    """Read a small RGB window from ``src`` using ``band_map`` ordering."""
 465    r = src.read(band_map[0], window=win)

 471def compute_block_minmax(src, band_map=(1, 2, 3), block=1024):
 472    """Scan an image in blocks to estimate robust min/max for each RGB band."""
 473    H, W = src.height, src.width
 474    r_vals, g_vals, b_vals = [], [], []
 475    for y in range(0, H, block):
 476        for x in range(0, W, block):
 477            win = Window(x, y, min(block, W - x), min(block, H - y))
 478            r, g, b = read_rgb_block(src, win, band_map)
 479            mask = None
 480            if src.nodata is not None:
 481                mask = (r == src.nodata) | (g == src.nodata) | (b == src.nodata)
 482            rl, rh = robust_minmax(r, mask)
 483            gl, gh = robust_minmax(g, mask)
 484            bl, bh = robust_minmax(b, mask)
 485            r_vals.append((rl, rh))
 486            g_vals.append((gl, gh))
 487            b_vals.append((bl, bh))
 488    r_lo = float(np.median([v[0] for v in r_vals]))
 489    r_hi = float(np.median([v[1] for v in r_vals]))
 490    g_lo = float(np.median([v[0] for v in g_vals]))
 491    g_hi = float(np.median([v[1] for v in g_vals]))
 492    b_lo = float(np.median([v[0] for v in b_vals]))
 493    b_hi = float(np.median([v[1] for v in b_vals]))
 494    return (r_lo, r_hi), (g_lo, g_hi), (b_lo, b_hi)

 500def get_mem_available_bytes() -> int:
 501    """Detect available memory from cgroup v2, cgroup v1, /proc/meminfo, or fallback.
 502
 503    Returns
 504    -------
 505    int
 506        Estimated available bytes. Falls back to 512 MB if detection fails.
 507    """
 508    # cgroup v2
 509    cgroup_max = Path("/sys/fs/cgroup/memory.max")
 510    cgroup_cur = Path("/sys/fs/cgroup/memory.current")
 511    try:
 512        if cgroup_max.exists() and cgroup_cur.exists():
 513            max_raw = cgroup_max.read_text().strip()
 514            if max_raw != "max":
 515                mem_max = int(max_raw)
 516                mem_cur = int(cgroup_cur.read_text().strip())
 517                return max(0, mem_max - mem_cur)
 518    except Exception:
 519        pass
 520
 521    # cgroup v1 (Docker pre-2020, some Kubernetes)
 522    cgroup_v1_limit = Path("/sys/fs/cgroup/memory/memory.limit_in_bytes")
 523    cgroup_v1_usage = Path("/sys/fs/cgroup/memory/memory.usage_in_bytes")
 524    try:
 525        if cgroup_v1_limit.exists() and cgroup_v1_usage.exists():
 526            limit = int(cgroup_v1_limit.read_text().strip())
 527            usage = int(cgroup_v1_usage.read_text().strip())
 528            # 9223372036854771712 is the sentinel "no limit" value in cgroup v1
 529            if limit < 9_000_000_000_000_000:
 530                return max(0, limit - usage)
 531    except Exception:
 532        pass
 533
 534    # /proc/meminfo (bare metal / macOS not available here)
 535    meminfo = Path("/proc/meminfo")
 536    try:
 537        if meminfo.exists():
 538            for line in meminfo.read_text().splitlines():
 539                if line.startswith("MemAvailable:"):
 540                    kb = int(line.split()[1])
 541                    return kb * 1024
 542    except Exception:
 543        pass
 544
 545    return 512 * 1024 * 1024  # conservative fallback

 548def choose_block_size_auto(
 549    out_width: int,
 550    out_height: int,
 551    requested: Optional[int],
 552    mem_budget_bytes: int,
 553) -> int:
 554    """Compute an optimal square block size fitting within ``mem_budget_bytes``.
 555
 556    Parameters
 557    ----------
 558    out_width, out_height
 559        Dimensions of the output raster in pixels.
 560    requested
 561        If provided and > 0, return it clamped to [256, 4096].
 562    mem_budget_bytes
 563        Available memory budget for block processing.
 564
 565    Returns
 566    -------
 567    int
 568        Block side length in pixels, multiple of 64, in [256, 4096].
 569
 570    Notes
 571    -----
 572    Bytes-per-pixel estimate (30):
 573    - band_a float32 (4) + band_b float32 (4)
 574    - den float32 (4) + v float32 (4) + t float32 (4)
 575    - gray uint16 (2) + rgba uint8×4 (4) + mask/overhead (~8)
 576    = 30 bytes/pixel
 577    Overhead factor 1.8 provides safety margin for temporaries and allocator.
 578    """
 579    if requested and requested > 0:
 580        return max(256, min(4096, int(requested)))
 581
 582    bytes_per_pixel = 30
 583    overhead_factor = 1.8
 584    usable = int(mem_budget_bytes / overhead_factor)
 585    max_pixels = max(1, usable // bytes_per_pixel)
 586    side = int(math.sqrt(max_pixels))
 587    side = max(256, min(4096, side, out_width, out_height))
 588    side = (side // 64) * 64
 589    return max(256, side)

 592def iter_windows(width: int, height: int, block_size: int):
 593    """Yield rasterio ``Window`` objects covering a raster in tiles.
 594
 595    Parameters
 596    ----------
 597    width, height
 598        Raster dimensions in pixels.
 599    block_size
 600        Tile side length in pixels.
 601
 602    Yields
 603    ------
 604    rasterio.windows.Window
 605        Non-overlapping windows covering the full extent.
 606    """
 607    for row in range(0, height, block_size):
 608        for col in range(0, width, block_size):
 609            w = min(block_size, width - col)
 610            h = min(block_size, height - row)
 611            yield Window.from_slices((row, row + h), (col, col + w))

 618def build_lut(cmap_name: str = "RdYlGn", levels: int = 65536) -> np.ndarray:
 619    """Build and cache a colormap LUT of shape (levels, 4) dtype uint8.
 620
 621    Parameters
 622    ----------
 623    cmap_name
 624        Matplotlib colormap name (e.g. "RdYlGn", "viridis").
 625    levels
 626        Number of discrete entries. 65536 eliminates banding for float32 index values.
 627
 628    Returns
 629    -------
 630    numpy.ndarray
 631        RGBA uint8 array of shape (levels, 4). Cached at module level.
 632    """
 633    key = (cmap_name, levels)
 634    if key not in _LUT_CACHE:
 635        import matplotlib  # noqa: PLC0415
 636
 637        cmap = matplotlib.colormaps.get_cmap(cmap_name)
 638        x = np.linspace(0.0, 1.0, levels, dtype=np.float32)
 639        _LUT_CACHE[key] = cmap(x, bytes=True)  # shape (levels, 4) uint8
 640    return _LUT_CACHE[key]

 668def srgb_to_linear(x: NDArray[np.floating]) -> NDArray[np.float32]:
 669    """
 670    Convert sRGB to linear RGB (gamma removal).
 671
 672    Parameters
 673    ----------
 674    x
 675        Array-like of sRGB values in [0, 1]. Any shape, last dimension length
 676        may be 3 for RGB but is not required. Values outside [0, 1] are not
 677        clamped here to avoid hiding data issues upstream.
 678
 679    Returns
 680    -------
 681    numpy.ndarray
 682        Linear RGB as float32, same shape as `x`.
 683
 684    Notes
 685    -----
 686    Uses the standard IEC 61966-2-1 transfer function. Thresholds and constants:
 687    - a = 0.055
 688    - thr = 0.04045
 689
 690    This function operates vectorized and preserves NaN locations.
 691
 692    Memory note
 693    -----------
 694    When ``x`` is already float32 the function operates **in-place** on the
 695    input array to avoid the peak of holding two full-size arrays in memory
 696    simultaneously (~2× reduction in memory spike for full-res images).
 697    The boolean masks ``low`` / ``~low`` partition the array disjointly, so
 698    modifying one subset never corrupts the values read by the other — the
 699    result is numerically identical to the copy-based path.
 700    """
 701    # In-place path: reuse the input buffer when dtype is already float32.
 702    # Non-float32 inputs still require a cast (unavoidable copy).
 703    a, thr = 0.055, 0.04045
 704    out = x if x.dtype == np.float32 else x.astype(np.float32)
 705    low = out <= thr
 706    out[low] /= 12.92
 707    out[~low] = ((out[~low] + a) / (1.0 + a)) ** 2.4
 708    return out

 711def linear_to_srgb(x: NDArray[np.floating]) -> NDArray[np.float32]:
 712    """
 713    Convert linear RGB to sRGB (gamma application).
 714
 715    Parameters
 716    ----------
 717    x
 718        Array-like of linear RGB values, typically in [0, 1]. Any shape.
 719
 720    Returns
 721    -------
 722    numpy.ndarray
 723        sRGB as float32 in [0, 1], same shape as `x`. Values are clipped to
 724        [0, 1] at the end.
 725
 726    Notes
 727    -----
 728    Uses the inverse IEC 61966-2-1 transfer function. Thresholds and constants:
 729    - a = 0.055
 730    - thr = 0.0031308
 731    """
 732    a, thr = 0.055, 0.0031308
 733    y = np.empty_like(x, dtype=np.float32)
 734    low = x <= thr
 735    y[low] = 12.92 * x[low]
 736    y[~low] = (1 + a) * (x[~low]) ** (1 / 2.4) - a
 737    return np.clip(y, 0.0, 1.0).astype(np.float32)

 740def gray_world(img_lin: NDArray[np.floating]) -> NDArray[np.float32]:
 741    """
 742    Apply Gray-World white balance to linear RGB.
 743
 744    Parameters
 745    ----------
 746    img_lin
 747        Linear RGB image as float array with last dimension size 3.
 748        Expected range [0, 1], NaN allowed (they are ignored in scaling).
 749
 750    Returns
 751    -------
 752    numpy.ndarray
 753        White-balanced linear RGB as float32, clipped to [0, 1].
 754
 755    Notes
 756    -----
 757    The Gray-World assumption scales each channel by the ratio of the mean of
 758    all channels to the channel mean: scale_c = mean(mean(R,G,B)) / mean(c).
 759    A small epsilon (1e-8) avoids division by zero.
 760    """
 761    if img_lin.shape[-1] != 3:
 762        raise ValueError("img_lin must have last dimension of size 3 (RGB).")
 763    means = img_lin.reshape(-1, 3).mean(axis=0)
 764    scale = means.mean() / (means + 1e-8)
 765    return np.clip(img_lin * scale, 0.0, 1.0).astype(np.float32)

 792def read_rgb(path: Path) -> NDArray[np.float32]:
 793    """
 794    Read an image and return sRGB float32 in [0, 1].
 795
 796    Parameters
 797    ----------
 798    path
 799        Path to PNG/JPG/TIFF. TIFF may be 8/16-bit and can carry NoData.
 800
 801    Returns
 802    -------
 803    numpy.ndarray
 804        Array of shape (H, W, 3) float32 in [0, 1]. Pixels matching NoData in
 805        any RGB band are returned as NaN triplets.
 806
 807    Behavior
 808    --------
 809    1) Try rasterio:
 810       - Read bands 1..3 as float32.
 811       - Scale by dtype max if > 1 (e.g., 65535 for uint16).
 812       - Apply NoData as NaN.
 813    2) Fallback to PIL:
 814       - Convert to 'RGB' and scale by 255.
 815
 816    Raises
 817    ------
 818    FileNotFoundError
 819        If the file does not exist.
 820    ValueError
 821        If a raster is opened but has fewer than 3 bands.
 822
 823    Notes
 824    -----
 825    The function prefers rasterio to preserve georeferencing metadata when
 826    available, though it returns only pixel data.
 827    """
 828    if not Path(path).exists():
 829        raise FileNotFoundError(f"File not found: {path}")
 830
 831    try:
 832        with rasterio.open(path) as src:
 833            if src.count < 3:
 834                raise ValueError("Raster has fewer than 3 bands.")
 835            r = src.read(1).astype(np.float32)
 836            g = src.read(2).astype(np.float32)
 837            b = src.read(3).astype(np.float32)
 838            maxv = _dtype_max(src.dtypes[0])
 839            if maxv > 1.0:
 840                r, g, b = r / maxv, g / maxv, b / maxv
 841            rgb = np.stack([r, g, b], axis=-1).astype(np.float32)
 842            if src.nodata is not None:
 843                nod = src.nodata
 844                mask = (r == nod) | (g == nod) | (b == nod)
 845                rgb[mask] = np.nan
 846            return rgb.astype(np.float32)
 847    except Exception:
 848        # Fallback for non-GDAL formats or simple images.
 849        im = Image.open(path).convert("RGB")
 850        return (np.asarray(im, dtype=np.float32) / 255.0).astype(np.float32)

 853def read_red_nir(
 854    src: DatasetReader,
 855    red_band: int = 3,
 856    nir_band: int = 4,
 857) -> Tuple[NDArray[np.float32], NDArray[np.float32]]:
 858    """
 859    Read and normalize RED and NIR bands from an open rasterio dataset.
 860
 861    Parameters
 862    ----------
 863    src
 864        Open rasterio dataset.
 865    red_band
 866        1-based index of the RED band (default 3).
 867    nir_band
 868        1-based index of the NIR band (default 4).
 869
 870    Returns
 871    -------
 872    (red, nir)
 873        Tuple of float32 arrays in [0, 1] when integer input types are used.
 874        NoData values are returned as NaN.
 875
 876    Raises
 877    ------
 878    ValueError
 879        If band indices are invalid or dataset has insufficient bands.
 880    """
 881    if src.count < max(red_band, nir_band):
 882        raise ValueError("Dataset does not contain the requested bands.")
 883    red = src.read(red_band).astype(np.float32)
 884    nir = src.read(nir_band).astype(np.float32)
 885    maxv = _dtype_max(src.dtypes[0])
 886    if maxv > 1.0:
 887        red, nir = red / maxv, nir / maxv
 888    if src.nodata is not None:
 889        m = (red == src.nodata) | (nir == src.nodata)
 890        red[m] = np.nan
 891        nir[m] = np.nan
 892    return red.astype(np.float32), nir.astype(np.float32)

 895def save_quick_preview(
 896    src_path: Path,
 897    out_path: Path,
 898    max_size: int = 1024,
 899) -> Tuple[int, int]:
 900    """Generate a fast RGB preview PNG from the source image.
 901
 902    Uses rasterio native ``out_shape`` downsampling when the source is a
 903    multi-band raster to avoid loading the full-resolution image into memory.
 904    For large ortofotos this is the difference between a ~10 MB allocation and
 905    a ~2 GB one.
 906
 907    Falls back to the original PIL-based path for formats rasterio cannot open
 908    or for single-band rasters without enough RGB bands.
 909
 910    Steps
 911    -----
 912    1. Attempt rasterio path: read bands 1-3 already at target resolution via
 913       bilinear resampling (no full-res array ever created).
 914    2. On any failure, fall back to ``read_rgb`` + PIL thumbnail (original path).
 915    3. Normalize to [0, 1], fill NaN/nodata with 0, scale to uint8, save PNG.
 916    """
 917    try:
 918        with rasterio.open(src_path) as src:
 919            if src.count >= 3:
 920                src_h, src_w = src.height, src.width
 921                scale = max_size / max(src_h, src_w, 1)
 922                if scale < 1.0:
 923                    out_h = max(1, int(round(src_h * scale)))
 924                    out_w = max(1, int(round(src_w * scale)))
 925                else:
 926                    out_h, out_w = src_h, src_w
 927                data = src.read(
 928                    [1, 2, 3],
 929                    out_shape=(3, out_h, out_w),
 930                    resampling=Resampling.bilinear,
 931                    masked=True,
 932                ).astype(np.float32)
 933                maxv = _dtype_max(src.dtypes[0])
 934                if maxv > 1.0:
 935                    data = data / maxv
 936                arr = np.moveaxis(np.ma.filled(data, 0.0), 0, -1)
 937                arr = np.nan_to_num(arr, nan=0.0)
 938                arr = np.clip(arr, 0.0, 1.0)
 939                im = Image.fromarray((arr * 255).astype(np.uint8))
 940                im.save(out_path, format="PNG")
 941                return im.size
 942    except Exception:
 943        pass
 944
 945    # Fallback: original full-res load + PIL thumbnail
 946    rgb = read_rgb(src_path)
 947    arr = np.nan_to_num(rgb, nan=0.0)
 948    arr = np.clip(arr, 0.0, 1.0)
 949    im = Image.fromarray((arr * 255).astype(np.uint8))
 950    w, h = im.size
 951    if max(w, h) > max_size:
 952        im.thumbnail((max_size, max_size))
 953    im.save(out_path, format="PNG")
 954    return im.size

 957def write_float32_geotiff(
 958    template: DatasetReader,
 959    arr: NDArray[np.floating],
 960    out_path: Path,
 961    desc: str,
 962) -> None:
 963    """
 964    Write a single-band float32 GeoTIFF using a template dataset profile.
 965
 966    Parameters
 967    ----------
 968    template
 969        Open rasterio dataset to copy georeferencing/profile from.
 970    arr
 971        2D array to write. NaNs are converted to nodata sentinel (-9999.0).
 972    out_path
 973        Output file path.
 974    desc
 975        Per-band description to set on the written band.
 976
 977    Notes
 978    -----
 979    - Uses DEFLATE compression, predictor=3 (floating point), tiled=True, and
 980      bigtiff="IF_SAFER" to mitigate file size and memory spikes.
 981    - This function writes count=1 with dtype=float32 and nodata=-9999.0.
 982    """
 983    profile = template.profile.copy()
 984    profile.update(
 985        count=1,
 986        dtype="float32",
 987        nodata=-9999.0,
 988        compress="deflate",
 989        predictor=3,
 990        tiled=True,
 991        bigtiff="IF_SAFER",
 992    )
 993    data = np.where(np.isfinite(arr), arr, -9999.0).astype(np.float32)
 994    with rasterio.open(out_path, "w", **profile) as dst:
 995        dst.write(data, 1)
 996        try:
 997            dst.set_band_description(1, desc)
 998        except Exception:
 999            # Some drivers may not support band descriptions.
1000            pass

1035def preprocess_rgb_once(
1036    input_path: Path,
1037    cfg: PreprocessConfig,
1038    progress_cb: Optional[Callable[[str, float, str], None]] = None,
1039) -> Tuple[
1040    NDArray[np.float32],
1041    Optional[Path],
1042    Optional[Path],
1043    Optional[Path],
1044    Optional[Path],
1045]:
1046    """
1047    Preprocess RGB exactly once and cache results to disk.
1048
1049    Steps
1050    -----
1051    1) Read sRGB [0, 1] and convert to linear RGB.
1052    2) Optional gray-world white balance.
1053    3) Optional shadow mask: pixels with low linear RGB sum are set to NaN.
1054    4) Cache artifacts:
1055       - NPY with exact linear float32 RGB (for scientific use).
1056       - PNG (sRGB 8-bit) preview for UI.
1057
1058    Parameters
1059    ----------
1060    input_path
1061        Path to the input image (PNG/JPG/TIFF).
1062    cfg
1063        PreprocessConfig controlling the operations and cache directory.
1064
1065    Returns
1066    -------
1067    (rgb_lin, png_path, npy_path, vi_gray_path, vi_heat_path, heatmap_path)
1068        - rgb_lin: linear RGB float32 en [0, 1] con NaNs donde aplique.
1069        - png_path: vista previa RGB normalizada en PNG, si se generó.
1070        - npy_path: cache NPY con los flotantes exactos, si se generó.
1071        - vi_gray_path: PNG en escala de grises con VI (G/R), si se generó.
1072        - vi_heat_path: PNG coloreado (tonos naranja/rojo) derivado del VI.
1073        - heatmap_path: PNG RGBA con heatmap pseudo-NDVI y máscara de vegetación.
1074
1075    Notes
1076    -----
1077    - If the NPY cache already exists, it is used directly (fast path). The
1078      preview PNG is not recomputed if already present.
1079    - NaN pixels are mapped to black (0) in the preview PNG to avoid visible
1080      checkerboards, but the NPY preserves NaNs exactly.
1081    """
1082    in_key = input_path.stem
1083    cache_dir = cfg.cache_dir or input_path.parent
1084    cache_dir.mkdir(parents=True, exist_ok=True)
1085    npy_path = cache_dir / f"{in_key}__rgb_preproc_linear.npy"
1086    png_path = cache_dir / f"{in_key}__rgb_preproc_preview.png"
1087
1088    def emit_progress(state: str, progress: float, message: str) -> None:
1089        if not progress_cb:
1090            return
1091        try:
1092            progress_cb(state, progress, message)
1093        except Exception:
1094            return
1095
1096    def _downsample_for_preview(
1097        rgb_data: NDArray[np.float32], max_dim: int
1098    ) -> NDArray[np.float32]:
1099        h, w = rgb_data.shape[:2]
1100        if max_dim <= 0 or max(h, w) <= max_dim:
1101            return rgb_data
1102        step = int(np.ceil(max(h, w) / max_dim))
1103        step = max(1, step)
1104        ds = rgb_data[::step, ::step]
1105        if ds.size == 0:
1106            return rgb_data
1107        # Ensure we do not exceed max_dim due to rounding.
1108        h2, w2 = ds.shape[:2]
1109        if max(h2, w2) > max_dim:
1110            step2 = int(np.ceil(max(h2, w2) / max_dim))
1111            step2 = max(1, step2)
1112            ds = ds[::step2, ::step2]
1113            if ds.size == 0:
1114                return rgb_data
1115        return ds

1117    def _apply_heatmap_colormap(
1118        norm: NDArray[np.float32], mask: Optional[NDArray[np.bool_]]
1119    ) -> NDArray[np.uint8]:
1120        stops = np.array([0.0, 0.5, 1.0], dtype=np.float32)
1121        colors = np.array(
1122            [
1123                [0.647, 0.0, 0.149],  # deep red
1124                [1.0, 0.973, 0.533],  # yellow
1125                [0.0, 0.4, 0.0],  # dark green
1126            ],
1127            dtype=np.float32,
1128        )
1129        flat = np.nan_to_num(norm.ravel(), nan=0.0)
1130        r = np.interp(flat, stops, colors[:, 0]).reshape(norm.shape)
1131        g = np.interp(flat, stops, colors[:, 1]).reshape(norm.shape)
1132        b = np.interp(flat, stops, colors[:, 2]).reshape(norm.shape)
1133        rgba = np.stack(
1134            [
1135                np.clip(r * 255.0, 0, 255).astype(np.uint8),
1136                np.clip(g * 255.0, 0, 255).astype(np.uint8),
1137                np.clip(b * 255.0, 0, 255).astype(np.uint8),
1138                np.full(norm.shape, 255, dtype=np.uint8),
1139            ],
1140            axis=-1,
1141        )
1142        if mask is not None:
1143            alpha = np.where(mask, 255, 0).astype(np.uint8)
1144            rgba[..., 3] = alpha
1145        return rgba
1146
1147    def ensure_rgb_preview() -> Optional[Path]:
1148        if png_path.exists():
1149            return png_path
1150        _png_tmp = png_path.with_suffix(".png.tmp")
1151        save_quick_preview(input_path, _png_tmp, max_size=cfg.preview_max_dim)
1152        os.replace(_png_tmp, png_path)
1153        return png_path
1154
1155    def ensure_vi_outputs(
1156        rgb_data: NDArray[np.float32],
1157    ) -> Tuple[Optional[Path], Optional[Path]]:
1158        vi_gray_path = cache_dir / f"{in_key}__vi_gr_ratio.png"
1159        vi_heat_path = cache_dir / f"{in_key}__vi_gr_heat.png"
1160        if vi_gray_path.exists() and vi_heat_path.exists():
1161            return vi_gray_path, vi_heat_path
1162
1163        vi_src = _downsample_for_preview(rgb_data, cfg.preview_max_dim)
1164
1165        g = vi_src[..., 1]
1166        r = vi_src[..., 0]
1167        vi_ratio = np.divide(g, r + 1e-6)
1168        vi_ratio[np.isnan(vi_src).any(axis=-1)] = np.nan
1169
1170        if not np.isfinite(vi_ratio).any():
1171            return None, None
1172
1173        lo = float(np.nanpercentile(vi_ratio, 5))
1174        hi = float(np.nanpercentile(vi_ratio, 95))
1175        if not np.isfinite(lo):
1176            lo = float(np.nanmin(vi_ratio))
1177        if not np.isfinite(hi):
1178            hi = float(np.nanmax(vi_ratio))
1179        if hi <= lo:
1180            hi = lo + 1e-3
1181        vi_norm = np.clip((vi_ratio - lo) / (hi - lo), 0.0, 1.0)
1182        vi_norm = np.nan_to_num(vi_norm, nan=0.0)
1183
1184        vi_gray = (vi_norm * 255.0).astype(np.uint8)
1185        Image.fromarray(vi_gray, mode="L").save(vi_gray_path)
1186
1187        heat = np.zeros((*vi_norm.shape, 3), dtype=np.float32)
1188        heat[..., 0] = vi_norm
1189        heat[..., 1] = np.clip(vi_norm * 0.6, 0.0, 1.0)
1190        heat[..., 2] = np.clip(vi_norm * 0.2, 0.0, 1.0)
1191        heat_img = (heat * 255.0).astype(np.uint8)
1192        Image.fromarray(heat_img, mode="RGB").save(vi_heat_path)
1193
1194        return vi_gray_path, vi_heat_path
1195
1196    def ensure_heatmap(rgb_data: NDArray[np.float32]) -> Optional[Path]:
1197        heatmap_path = cache_dir / f"{in_key}__vi_heatmap.png"
1198        if heatmap_path.exists():
1199            return heatmap_path
1200
1201        heat_src = _downsample_for_preview(rgb_data, cfg.preview_max_dim)
1202        R = heat_src[..., 0]
1203        G = heat_src[..., 1]
1204        B = heat_src[..., 2]
1205        EPS = 1e-6
1206
1207        ngrdi = (G - R) / (G + R + EPS)
1208        vari = (G - R) / (G + R - B + EPS)
1209        gli = (2.0 * G - R - B) / (2.0 * G + R + B + EPS)
1210        exg = 2.0 * G - R - B
1211        pseudo = 0.4 * ngrdi + 0.25 * vari + 0.25 * gli + 0.10 * exg
1212        pseudo = np.clip(pseudo, -1.0, 1.0).astype(np.float32)
1213
1214        vi = G / (R + EPS)
1215        mask = (exg > 0.05) & (ngrdi > 0.0) & (vi > 1.2)
1216
1217        norm = (pseudo + 1.0) * 0.5
1218        norm = np.clip(norm, 0.0, 1.0).astype(np.float32)
1219
1220        rgba = _apply_heatmap_colormap(norm, mask)
1221        try:
1222            Image.fromarray(rgba, mode="RGBA").save(heatmap_path)
1223        except Exception:
1224            return None
1225        return heatmap_path
1226
1227    # Fast path: todos los artefactos ya existen — nada que generar.
1228    _vi_gray_maybe = cache_dir / f"{in_key}__vi_gr_ratio.png"
1229    _vi_heat_maybe = cache_dir / f"{in_key}__vi_gr_heat.png"
1230    _heatmap_maybe = cache_dir / f"{in_key}__vi_heatmap.png"

1330def generate_nd_index_rgba(
1331    source_path: Path,
1332    cache_dir: Path,
1333    asset_uuid: str,
1334    band_a: int = 1,
1335    band_b: int = 2,
1336    cmap_name: str = "RdYlGn",
1337    lut_levels: int = 65536,
1338    mem_budget_bytes: Optional[int] = None,
1339    max_out_dim: int = 2048,
1340) -> Optional[Path]:
1341    """Generate a normalized-difference spectral index RGBA PNG.
1342
1343    Computes ``(band_b - band_a) / (band_b + band_a)``, normalises to [0, 1],
1344    applies a colormap LUT, and saves a transparent RGBA PNG where NoData pixels
1345    have alpha = 0.
1346
1347    Memory strategy
1348    ---------------
1349    The source is read **at the output resolution** using rasterio's native
1350    ``out_shape`` parameter (bilinear resampling).  For a 10 000 × 8 000 px
1351    ortofoto with ``max_out_dim=2048`` the two band arrays together occupy
1352    only ~32 MB instead of ~640 MB.  The final RGBA array is ~16 MB.
1353    The result is saved with Pillow, which streams PNG to disk without
1354    buffering the full raster in the GDAL PNG driver.
1355
1356    Parameters
1357    ----------
1358    source_path
1359        Absolute path to the input raster (GeoTIFF, PNG, JPG).
1360    cache_dir
1361        Directory where the output PNG will be stored.
1362    asset_uuid
1363        UUID of the asset — used to name the output file.
1364    band_a
1365        1-based index of band A (subtracted term). Default 1.
1366    band_b
1367        1-based index of band B (added term). Default 2.
1368        True NDVI: band_a=3 (RED), band_b=4 (NIR).
1369        G/R ratio: band_a=1 (R), band_b=2 (G).
1370    cmap_name
1371        Matplotlib colormap name. Default "RdYlGn".
1372    lut_levels
1373        LUT resolution. 65 536 eliminates banding for float32 index values.
1374    mem_budget_bytes
1375        Kept for API compatibility; not used in this implementation.
1376    max_out_dim
1377        Maximum width or height of the output PNG in pixels. Default 2048.
1378
1379    Returns
1380    -------
1381    Path or None
1382        Path to the generated PNG, or None if generation failed.
1383
1384    Notes
1385    -----
1386    - Cached: if the output PNG already exists, it is returned immediately.
1387    - Atomic write: ``.tmp`` → ``os.replace`` (no partial files on crash).
1388    - Formula: ``v = (b - a) / (b + a)``; zero denominator and non-finite
1389      values are treated as nodata (alpha = 0).
1390    """
1391    out_path = cache_dir / f"{asset_uuid}__nd_index_rgba.png"
1392    if out_path.exists():
1393        return out_path
1394
1395    cache_dir.mkdir(parents=True, exist_ok=True)
1396    lut = build_lut(cmap_name, lut_levels)
1397    # Unique tmp path prevents collision when agrovista-meta on-demand call and
1398    # background task run concurrently (both would write to the same .png.tmp).
1399    tmp_path = out_path.parent / f".{out_path.name}.{os.getpid()}.tmp"
1400
1401    try:
1402        with rasterio.open(source_path) as src:
1403            n_bands = src.count
1404
1405            # Auto-select bands when the requested bands exceed available count
1406            eff_a, eff_b = band_a, band_b
1407            if n_bands < max(eff_a, eff_b):
1408                if n_bands >= 2:
1409                    eff_a, eff_b = 1, 2
1410                elif n_bands == 1:
1411                    eff_a, eff_b = 1, 1
1412                else:
1413                    return None
1414
1415            src_h, src_w = src.height, src.width
1416            src_dtype = src.dtypes[0] if n_bands > 0 else "uint8"
1417
1418            # Compute output dimensions capped at max_out_dim
1419            scale = max_out_dim / max(src_h, src_w, 1)
1420            if scale < 1.0:
1421                out_h = max(1, int(round(src_h * scale)))
1422                out_w = max(1, int(round(src_w * scale)))
1423            else:
1424                out_h, out_w = src_h, src_w
1425
1426            # Read both bands at output resolution — bounded memory
1427            # e.g. 2048×2048×2 bands×4 bytes = ~32 MB
1428            data = src.read(
1429                [eff_a, eff_b],
1430                out_shape=(2, out_h, out_w),
1431                resampling=Resampling.bilinear,
1432                masked=True,
1433            ).astype(np.float32)
1434
1435        maxv = _dtype_max(src_dtype)
1436        if maxv > 1.0:
1437            data = data / maxv
1438
1439        a_ma = data[0]
1440        b_ma = data[1]
1441        del data  # free the 2-band array
1442
1443        base_mask = np.ma.getmaskarray(a_ma) | np.ma.getmaskarray(b_ma)
1444        a_f = np.ma.filled(a_ma, 0.0)
1445        b_f = np.ma.filled(b_ma, 0.0)
1446        del a_ma, b_ma
1447
1448        # Normalised difference: (b - a) / (b + a)
1449        den = a_f + b_f
1450        v = np.empty_like(den, dtype=np.float32)
1451        np.subtract(b_f, a_f, out=v)
1452        np.divide(v, den, out=v, where=(den != 0.0))
1453        del a_f, b_f, den
1454
1455        invalid = ~np.isfinite(v) | (v == 0.0)
1456        mask = base_mask | invalid
1457        del base_mask, invalid
1458
1459        # Normalise [-1, 1] → [0, 1]
1460        t = (v + 1.0) * 0.5
1461        np.clip(t, 0.0, 1.0, out=t)
1462        t[mask] = 0.0
1463        del v
1464
1465        # Map to LUT → RGBA uint8
1466        gray = np.rint(t * (lut_levels - 1)).astype(np.uint16, copy=False)
1467        del t
1468        rgba = lut[gray].copy()  # (out_h, out_w, 4) uint8
1469        del gray
1470
1471        # Apply nodata transparency
1472        rgba[mask, 3] = 0
1473        del mask
1474
1475        # Save via Pillow (streams PNG without buffering full raster in GDAL)
1476        try:
1477            Image.fromarray(rgba, mode="RGBA").save(tmp_path, format="PNG")
1478        except Exception:
1479            try:
1480                tmp_path.unlink(missing_ok=True)
1481            except Exception:
1482                pass
1483            raise
1484
1485        # Atomic rename
1486        os.replace(tmp_path, out_path)
1487
1488    except Exception:
1489        try:
1490            current_app.logger.exception(
1491                "media: generate_nd_index_rgba failed for %s", asset_uuid
1492            )
1493        except Exception:
1494            pass
1495        try:
1496            tmp_path.unlink(missing_ok=True)
1497        except Exception:
1498            pass
1499        return None
1500
1501    return out_path

1700def compute_wb_factors_from_tif(
1701    tif_path: Path,
1702    block_size: int = 1024,
1703    sample_step: int = 4,
1704) -> Dict[str, float]:
1705    """Compute gray-world white balance scale factors by block-sampling a GeoTIFF.
1706
1707    Reads every ``sample_step``-th block of ``block_size`` × ``block_size`` pixels
1708    from bands 1-3, accumulates per-channel means, and returns the gray-world
1709    correction factors (global_mean / channel_mean) as plain Python floats so the
1710    result is directly JSON-serializable.
1711
1712    Returns {"scale_r": float, "scale_g": float, "scale_b": float}.
1713    Identity dict (all 1.0) is returned for images with fewer than 3 bands or
1714    when all sampled pixels are black/NoData.
1715    """
1716    with rasterio.open(str(tif_path)) as ds:
1717        if ds.count < 3:
1718            return {"scale_r": 1.0, "scale_g": 1.0, "scale_b": 1.0}
1719        h, w = ds.height, ds.width
1720        raw_dtype = np.dtype(ds.dtypes[0])
1721        dtype_max = float(np.iinfo(raw_dtype).max) if np.issubdtype(raw_dtype, np.integer) else 1.0
1722
1723        sum_r = sum_g = sum_b = 0.0
1724        n = 0
1725        stride = block_size * sample_step
1726        for y in range(0, h, stride):
1727            for x in range(0, w, stride):
1728                win = Window(x, y, min(block_size, w - x), min(block_size, h - y))
1729                data = ds.read([1, 2, 3], window=win).astype(np.float32) / dtype_max
1730                valid = np.any(data > 0.0, axis=0)
1731                if not valid.any():
1732                    continue
1733                sum_r += float(data[0][valid].mean())
1734                sum_g += float(data[1][valid].mean())
1735                sum_b += float(data[2][valid].mean())
1736                n += 1
1737
1738    if n == 0 or sum_r == 0.0 or sum_g == 0.0 or sum_b == 0.0:
1739        return {"scale_r": 1.0, "scale_g": 1.0, "scale_b": 1.0}
1740
1741    mean_r = sum_r / n
1742    mean_g = sum_g / n
1743    mean_b = sum_b / n
1744    global_mean = (mean_r + mean_g + mean_b) / 3.0
1745
1746    return {
1747        "scale_r": float(global_mean / mean_r),
1748        "scale_g": float(global_mean / mean_g),
1749        "scale_b": float(global_mean / mean_b),
1750    }

1753def read_tif_window_as_linear_rgb(
1754    tif_path: Path,
1755    x0: int,
1756    y0: int,
1757    x1: int,
1758    y1: int,
1759    wb_factors: Optional[Dict[str, float]] = None,
1760) -> NDArray[np.float32]:
1761    """Read a pixel BBox from a GeoTIFF as linear float32 RGB in [0, 1].
1762
1763    Bands 1-3 are read into shape (y1-y0, x1-x0, 3), converted from sRGB to
1764    linear light, and optionally scaled by ``wb_factors`` (gray-world correction).
1765    NoData/out-of-range values are clipped to [0, 1] after each transform.
1766    """
1767    with rasterio.open(str(tif_path)) as ds:
1768        win = Window(x0, y0, x1 - x0, y1 - y0)
1769        raw_dtype = np.dtype(ds.dtypes[0])
1770        dtype_max = float(np.iinfo(raw_dtype).max) if np.issubdtype(raw_dtype, np.integer) else 1.0
1771        data = ds.read([1, 2, 3], window=win).astype(np.float32) / dtype_max  # (3, H, W)
1772
1773    rgb = np.moveaxis(data, 0, -1)  # (H, W, 3)
1774    np.clip(rgb, 0.0, 1.0, out=rgb)
1775
1776    rgb = srgb_to_linear(rgb)
1777
1778    if wb_factors:
1779        rgb[:, :, 0] *= float(wb_factors.get("scale_r", 1.0))
1780        rgb[:, :, 1] *= float(wb_factors.get("scale_g", 1.0))
1781        rgb[:, :, 2] *= float(wb_factors.get("scale_b", 1.0))
1782        np.clip(rgb, 0.0, 1.0, out=rgb)
1783
1784    return rgb
